    return {

    sectionQuestions() {
      const section = this.sections.find(
        s => s.id === this.currentSection
      )
      return section ? section.questions : []
    },

    currentQuestion() {
      if (!this.sectionQuestions.length) return null
      return this.sectionQuestions[this.currentQuestionIndex] || null
    },

    options() {
      if (!this.currentQuestion) return []

      return [
        { key: "A", value: this.currentQuestion.option_a },
        { key: "B", value: this.currentQuestion.option_b },
        { key: "C", value: this.currentQuestion.option_c },
        { key: "D", value: this.currentQuestion.option_d }
      ]
    },

    formattedTime() {
      const minutes = Math.floor(this.timeLeft / 60)
      const seconds = this.timeLeft % 60
      return `${minutes}:${seconds.toString().padStart(2, "0")}`
    }
  },

  methods: {
    async startExam() {
  try {
    const res = await fetch(
      `http://localhost:8000/api/exam/${this.examId}/start`,
      { method: "POST" }
    )

    const data = await res.json()

    this.exam = data.exam
    this.sections = data.exam.sections
    this.attemptId = data.attempt_id

    // Flatten all questions into a single array
    this.questions = this.exam.sections.flatMap(section =>
      (section.questions || []).map(q => ({
        id: q.id,
        text: q.question_text,
        marks: q.marks,
        difficulty: q.difficulty,
        section: section.name,   // ensure this matches your currentSection
        section_id: section.id,
        options: [
          { letter: 'A', text: q.option_a },
          { letter: 'B', text: q.option_b },
          { letter: 'C', text: q.option_c },
          { letter: 'D', text: q.option_d }
        ]
      }))
    )

    this.currentSection = this.sections[0]?.id || null
    this.timeLeft = this.exam.duration * 60

    this.startTimer()
  } catch (err) {
    console.error(err)
  } finally {
    this.loading = false
  }
},

    changeSection(sectionId) {
      this.currentSection = sectionId
      this.currentQuestionIndex = 0
    },

    nextQuestion() {
      if (this.currentQuestionIndex < this.sectionQuestions.length - 1) {
        this.currentQuestionIndex++
      }
    },

      if (this.currentQuestionIndex > 0) {
        this.currentQuestionIndex--
      }
    },

      if (!this.visitedQuestions.includes(questionId)) {
        this.visitedQuestions.push(questionId)

            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({

              selected_option: option
            })
          }
        )
      } catch (err) {
        console.error(err)
      }
    },

        )
      } catch (err) {
        console.error(err)
      }
    }
  },

    this.startExam()
  }

@router.post("/attempt/{attempt_id}/answer")
def save_answer(attempt_id: int, payload: Dict, db: Session = Depends(get_db)):
    question_id = payload.get("question_id")
    selected_option = payload.get("selected_option")

    if not question_id or selected_option not in ["A", "B", "C", "D"]:
        raise HTTPException(status_code=400, detail="Invalid question_id or selected_option")

    attempt = db.query(exam_attempt_models.ExamAttemptModel).filter(
        exam_attempt_models.ExamAttemptModel.id == attempt_id
    ).first()

    if not attempt:
        raise HTTPException(status_code=404, detail="Attempt not found")

    if attempt.completed_at:
        raise HTTPException(status_code=400, detail="Exam already submitted")

    answer = db.query(exam_attempt_models.StudentAnswerModel).filter(
        exam_attempt_models.StudentAnswerModel.attempt_id == attempt_id,
        exam_attempt_models.StudentAnswerModel.question_id == question_id
    ).first()

    if answer:
        answer.selected_answer = selected_option
    else:
        answer = exam_attempt_models.StudentAnswerModel(
            attempt_id=attempt_id,
            question_id=question_id,
            selected_answer=selected_option
        )
        db.add(answer)

    db.commit()
    return {"message": "Answer saved successfully"}

@router.post("/attempt/{attempt_id}/submit")
def submit_exam(attempt_id: int, db: Session = Depends(get_db)):
    attempt = db.query(exam_attempt_models.ExamAttemptModel).filter(
        exam_attempt_models.ExamAttemptModel.id == attempt_id
    ).first()

    if not attempt:
        raise HTTPException(status_code=404, detail="Attempt not found")

    if attempt.completed_at:
        raise HTTPException(status_code=400, detail="Exam already submitted")

    answers = db.query(exam_attempt_models.StudentAnswerModel).filter(
        exam_attempt_models.StudentAnswerModel.attempt_id == attempt_id
    ).all()

    score = 0
    total_marks = 0
    for ans in answers:
        question = db.query(question_models.QuestionModel).filter(
            question_models.QuestionModel.id == ans.question_id
        ).first()
        if not question:
            continue
        total_marks += question.marks
        if ans.selected_answer == question.correct_answer:
            score += question.marks

    attempt.score = score
    attempt.completed_at = datetime.utcnow()
    db.commit()

    percentage = (score / total_marks * 100) if total_marks > 0 else 0
    return {
        "score": score, 
        "total_marks": total_marks, 
        "percentage": percentage,
        "message": "Exam submitted successfully"
